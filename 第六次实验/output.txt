int main() {
    int a;
    int b;
    int c;
    a = 10;
    b = 20;
    c = a + b * 2;
    if (c > 30) {
        print c;
    } else {
        print a;
    }
    while (a > 0) {
        a = a - 1;
    }
    return 0;
}
